# Find a column name by regex pattern (case-insensitive)
pick_col <- function(df, pattern, required = TRUE){
  nms <- names(df)
  hit <- nms[grepl(pattern, nms, ignore.case = TRUE)]
  if(length(hit) == 0){
    if(required) stop(paste("Could not find a column matching pattern:", pattern))
    return(NA_character_)
  }
  hit[1]
}

# Convert low-frequency series (monthly/quarterly) to weekly using LOCF on weekly date index
to_weekly_locf <- function(weekly_index, lowfreq_df, value_col) {

  lowfreq_df_clean <- lowfreq_df %>%
    dplyr::select(date, !!rlang::sym(value_col)) %>%
    dplyr::group_by(date) %>%                 # ← CRITICAL
    dplyr::summarise(
      value = mean(.data[[value_col]], na.rm = TRUE),
      .groups = "drop"
    )

  weekly_index %>%
    dplyr::left_join(lowfreq_df_clean, by = "date") %>%
    dplyr::arrange(date) %>%
    dplyr::mutate(
      value = zoo::na.locf(value, na.rm = FALSE)
    ) %>%
    dplyr::rename(!!value_col := value)
}

# Aggregate any weekly data frame to month/quarter by mean
to_monthly_mean <- function(df, value_col) {
  df %>%
    dplyr::mutate(date = lubridate::floor_date(date, "month")) %>%
    dplyr::group_by(date) %>%
    dplyr::summarise(
      !!value_col := mean(.data[[value_col]], na.rm = TRUE),
      .groups = "drop"
    ) %>%
    dplyr::arrange(date)
}

to_quarterly_mean <- function(df, value_cols){
  df %>%
    dplyr::mutate(date = lubridate::floor_date(date, "quarter")) %>%
    dplyr::group_by(date) %>%
    dplyr::summarise(across(all_of(value_cols), ~mean(.x, na.rm = TRUE)), .groups = "drop") %>%
    dplyr::arrange(date)
}

# Pairwise Granger (x -> y) on first differences
granger_pairwise <- function(df, y, x, lags){
  d <- df %>%
    arrange(date) %>%
    mutate(
      dy = .data[[y]] - lag(.data[[y]]),
      dx = .data[[x]] - lag(.data[[x]])
    ) %>%
    drop_na(dy, dx)
  
  gt <- lmtest::grangertest(dy ~ dx, order = lags, data = d)
  p <- gt$`Pr(>F)`[2]
  tibble(predictor = x, lags = lags, p_value = p)
}

# Joint Granger using VAR and vars::causality:
# tests whether ALL predictors jointly Granger-cause y.
granger_joint <- function(df, y, xs, lags){

  # First differences for stationarity
  d <- df %>%
    dplyr::arrange(date) %>%
    dplyr::mutate(
      dplyr::across(
        dplyr::all_of(c(y, xs)),
        ~ .x - dplyr::lag(.x),
        .names = "d_{.col}"
      )
    ) %>%
    tidyr::drop_na()

  dy  <- paste0("d_", y)
  dxs <- paste0("d_", xs)

  # VAR needs at least y + one predictor
  var_df <- d %>%
    dplyr::select(dplyr::all_of(c(dy, dxs)))

  if (ncol(var_df) < 2) {
    stop("Need at least y and one predictor for joint test.")
  }

  fit <- vars::VAR(var_df, p = lags, type = "const")

  caus <- vars::causality(fit, cause = dxs)

  tibble::tibble(
    test    = "Joint (all predictors)",
    lags    = lags,
    p_value = caus$Granger$p.value
  )
}

run_adf <- function(df, var_names, freq_label) {   # renamed vars → var_names
  purrr::map_dfr(var_names, function(v) {           # updated here too
    series <- df[[v]]
    series <- series[!is.na(series)]
    
    adf_level <- tryCatch(
      tseries::adf.test(series),
      error = function(e) list(p.value = NA)
    )
    
    adf_diff <- tryCatch(
      tseries::adf.test(diff(series)),
      error = function(e) list(p.value = NA)
    )
    
    tibble::tibble(
      frequency             = freq_label,
      variable              = v,
      adf_level_p           = round(adf_level$p.value, 4),
      adf_diff_p            = round(adf_diff$p.value, 4),
      stationary_at_level   = adf_level$p.value < 0.05,
      stationary_after_diff = adf_diff$p.value < 0.05
    )
  })
}

# ── Step 2: Lag Selection ─────────────────────────────────────────────────────
run_lag_selection <- function(df, y, xs, freq_label, max_lag = 8) {
  
  xs_avail <- xs[sapply(xs, function(x) sum(!is.na(df[[x]])) > max_lag * 2)]
  
  d <- df %>%
    dplyr::arrange(date) %>%
    dplyr::mutate(
      dplyr::across(dplyr::all_of(c(y, xs_avail)), ~ .x - dplyr::lag(.x), .names = "d_{.col}")
    ) %>%
    tidyr::drop_na()
  
  var_df <- d %>% dplyr::select(dplyr::starts_with("d_"))
  
  sel <- vars::VARselect(var_df, lag.max = max_lag, type = "const")
  
  cat("\n══ STEP 2: Lag Selection —", freq_label, "══\n")
  cat("AIC suggests lag:", sel$selection["AIC(n)"], "\n")
  cat("BIC suggests lag:", sel$selection["SC(n)"],  "\n")
  print(sel$criteria)
  
  as.integer(sel$selection["SC(n)"])
}

# ── Step 3: Pairwise Granger ──────────────────────────────────────────────────
run_pairwise <- function(df, y, xs, lags, freq_label) {
  
  cat("\n══ STEP 3: Pairwise Granger —", freq_label, "(lag =", lags, ") ══\n")
  cat("H0: predictor does NOT Granger-cause", y, "\n\n")
  
  results <- purrr::map_dfr(xs, function(x) {
    
    d <- df %>%
      dplyr::arrange(date) %>%
      dplyr::mutate(
        dy = .data[[y]] - dplyr::lag(.data[[y]]),
        dx = .data[[x]] - dplyr::lag(.data[[x]])
      ) %>%
      tidyr::drop_na(dy, dx)
    
    if (nrow(d) < lags * 3) {
      return(tibble::tibble(frequency = freq_label, predictor = x,
                            lag = lags, f_stat = NA, p_value = NA,
                            significant = "insufficient data"))
    }
    
    gt <- tryCatch(
      lmtest::grangertest(dy ~ dx, order = lags, data = d),
      error = function(e) NULL
    )
    
    if (is.null(gt)) {
      return(tibble::tibble(frequency = freq_label, predictor = x,
                            lag = lags, f_stat = NA, p_value = NA,
                            significant = "error"))
    }
    
    p <- gt$`Pr(>F)`[2]
    f <- gt$F[2]
    sig <- if (p < 0.01) "*** (1%)" else if (p < 0.05) "**  (5%)" else if (p < 0.10) "*   (10%)" else "not significant"
    
    tibble::tibble(
      frequency   = freq_label,
      predictor   = x,
      lag         = lags,
      f_stat      = round(f, 3),
      p_value     = round(p, 4),
      significant = sig
    )
  })
  
  print(results)
  results
}

# ── Step 4: Joint Granger ─────────────────────────────────────────────────────
run_joint <- function(df, y, xs, lags, freq_label) {
  
  cat("\n══ STEP 4: Joint Granger —", freq_label, "(lag =", lags, ") ══\n")
  cat("H0: ALL predictors jointly do NOT Granger-cause", y, "\n\n")
  
  xs_avail <- xs[sapply(xs, function(x) sum(!is.na(df[[x]])) > lags * 2)]
  if (length(xs_avail) == 0) { cat("No predictors available.\n"); return(NULL) }
  
  d <- df %>%
    dplyr::arrange(date) %>%
    dplyr::mutate(dplyr::across(
      dplyr::all_of(c(y, xs_avail)),
      ~ .x - dplyr::lag(.x), .names = "d_{.col}"
    )) %>%
    tidyr::drop_na()
  
  var_df <- d %>% dplyr::select(dplyr::all_of(paste0("d_", c(y, xs_avail))))
  
  fit <- tryCatch(
    vars::VAR(var_df, p = lags, type = "const"),
    error = function(e) { cat("VAR failed:", e$message, "\n"); NULL }
  )
  if (is.null(fit)) return(NULL)
  
  caus <- tryCatch(
    vars::causality(fit, cause = paste0("d_", xs_avail)),
    error = function(e) { cat("Causality test failed:", e$message, "\n"); NULL }
  )
  if (is.null(caus)) return(NULL)
  
  p_val   <- unname(as.vector(caus$Granger$p.value))[1]
  chi_val <- unname(as.vector(caus$Granger$statistic))[1]
  df_val  <- unname(as.vector(caus$Granger$parameter))[1]
  sig     <- if (p_val < 0.01) "*** (1%)" else if (p_val < 0.05) "**  (5%)" else if (p_val < 0.10) "*   (10%)" else "not significant"
  
  result <- tibble::tibble(
    frequency       = freq_label,
    test            = "Joint (all predictors → price)",
    predictors_used = paste(xs_avail, collapse = ", "),
    lag             = lags,
    chi_sq          = round(chi_val, 3),
    df_test         = df_val,
    p_value         = round(p_val, 4),
    significant     = sig
  )
  print(result)
  result
}

# ── Step 5: Diagnostics ───────────────────────────────────────────────────────
run_diagnostics <- function(df, y, xs, lags, freq_label) {
  
  cat("\n══ STEP 5: Residual Diagnostics —", freq_label, "══\n")
  
  xs_avail <- xs[sapply(xs, function(x) sum(!is.na(df[[x]])) > lags * 2)]
  
  d <- df %>%
    dplyr::arrange(date) %>%
    dplyr::mutate(
      dplyr::across(
        dplyr::all_of(c(y, xs_avail)),
        ~ .x - dplyr::lag(.x),
        .names = "d_{.col}"
      )
    ) %>%
    tidyr::drop_na()
  
  var_df <- d %>%
    dplyr::select(dplyr::all_of(paste0("d_", c(y, xs_avail))))
  
  fit <- tryCatch(
    vars::VAR(var_df, p = lags, type = "const"),
    error = function(e) { cat("VAR failed:", e$message, "\n"); NULL }
  )
  if (is.null(fit)) { cat("VAR failed, skipping diagnostics.\n"); return(NULL) }
  
  # Serial correlation test
  tryCatch({
    serial <- vars::serial.test(fit, lags.pt = lags + 2, type = "PT.asymptotic")
    p_serial <- unname(as.vector(serial$serial$p.value))[1]
    cat("Portmanteau test p-value:", round(p_serial, 4),
        ifelse(p_serial > 0.05,
               "✓ No autocorrelation in residuals",
               "✗ Autocorrelation detected — consider more lags"), "\n")
  }, error = function(e) {
    cat("Serial test failed (likely singular matrix — common with small quarterly samples):", 
        e$message, "\n")
  })
  
  # Normality test
  tryCatch({
    norm <- vars::normality.test(fit)
    p_norm <- unname(as.vector(norm$jb.mul$JB$p.value))[1]
    cat("Normality test (JB) p-value:", round(p_norm, 4), "\n")
  }, error = function(e) {
    cat("Normality test failed:", e$message, "\n")
  })
  
  invisible(fit)
}

Granger Causality Analysis: Determinants of UK Fruit & Vegetable Price Movements

Overview

This section tests whether macroeconomic and agricultural cost variables Granger-cause movements in UK retail fruit and vegetable prices. Granger causality tests whether the past values of a predictor variable contain statistically significant information about the future values of a target variable, beyond what the target’s own history already provides. A significant result does not imply structural causation — it indicates that the predictor has predictive content for the outcome. Tests were conducted at three temporal resolutions — weekly, monthly, and quarterly — and in two modes: pairwise (each predictor tested independently) and joint (all predictors tested simultaneously via a Vector Autoregression). The predictors examined are fuel price, Consumer Price Index (CPI), Agricultural Price Index (API), fertiliser price, and the Services Producer Price Index (SPPI).

Methodology

Prior to Granger testing, all series were assessed for stationarity using the Augmented Dickey-Fuller (ADF) test at both levels and first differences. Most series were found to be integrated of order one (I(1)) — non-stationary in levels but stationary after first differencing — satisfying the standard precondition for Granger testing in a VAR framework. One exception was fuel price at the weekly frequency, which was found to be I(2), requiring a second difference to achieve stationarity. Accordingly, the second-differenced series (fuel_price_d2) was substituted in weekly-level tests only. Fertiliser data was unavailable prior to 2017, resulting in partial coverage at monthly and quarterly frequencies.

Optimal lag lengths were selected using VARselect applied to first-differenced data, with the Bayesian Information Criterion (BIC) used as the primary selection criterion given its stronger penalty for over-parameterisation. BIC selected lag 1 at both weekly and monthly frequencies, and lag 4 at quarterly. However, the monthly lag was overridden to 3 (per AIC) after the BIC-selected model produced serially correlated residuals (Portmanteau p = 0.042). The quarterly lag was capped at 2 due to sample size constraints — with only 45 observations, a lag-4 VAR across six variables is severely over-parameterised and produces a singular coefficient matrix.

Pairwise Granger Causality Results

CPI is the dominant predictor and the only variable to achieve significance across multiple frequencies. It reaches significance at the monthly level (F = 2.90, p = 0.038) and strong significance at the quarterly level (F = 6.63, p = 0.004). This pattern suggests that inflationary pressures propagate into retail food prices with a lag of one to two quarters, consistent with the broader macroeconomic literature on price transmission. The quarterly result in particular is robust — a p-value of 0.004 at lag 2 provides strong evidence that CPI movements reliably precede fruit and vegetable price changes at that horizon.

Fuel price shows a marginally significant effect at the weekly frequency only (F = 3.08, p = 0.094), using the second-differenced series. This signal does not persist at monthly or quarterly frequencies, suggesting that any fuel price transmission into food prices is confined to very short-run, week-to-week fluctuations and dissipates quickly. This may reflect the role of fuel costs in immediate distribution and logistics, which are absorbed or passed through within days rather than propagating into longer-run price setting.

SPPI shows a borderline monthly effect (F = 2.13, p = 0.100) that warrants cautious interpretation. It may reflect indirect cost pressures through logistics and distribution services, but the evidence sits at the threshold of conventional significance and should not be overstated.

Neither the Agricultural Price Index nor fertiliser prices exhibit significant Granger causality at any frequency or lag specification. This is a notable null result. Despite the intuitive prior that upstream input costs should drive retail food prices, there is no statistical evidence of a predictive relationship in this dataset over the sample period. This may reflect the complexity of agricultural supply chains, producer hedging behaviour, or the dominance of imported produce in UK retail, where domestic input cost structures are less directly relevant.

Joint Granger Causality Results

Neither the monthly (χ² = 1.18, p = 0.284) nor the quarterly (χ² = 1.02, p = 0.429) joint tests reach significance. This apparent contradiction with the pairwise CPI results is a well-documented phenomenon in Granger causality analysis: joint VAR tests carry substantially more degrees of freedom — 15 for monthly and 10 for quarterly — which reduces statistical power when only one predictor in the system is genuinely informative while the rest are noise. The non-significance of the joint test does not invalidate the pairwise CPI finding; it reflects the diluting effect of jointly testing multiple null predictors alongside a real signal.

The weekly joint VAR could not be estimated due to matrix singularity. SPPI, filled via last-observation-carried-forward from quarterly data, produces a constant series within each quarter and collapses to zero after first differencing, rendering the covariance matrix rank-deficient. This predictor was excluded from the weekly joint test, but the remaining variables still produced a singular matrix at lag 1, likely due to near-perfect collinearity among the short differenced series. The weekly frequency is therefore best analysed through pairwise tests only.

Residual Diagnostics

The monthly VAR at lag 3 passes the Portmanteau test for serial correlation, confirming that the override from BIC’s lag-1 selection was appropriate. Non-normality of residuals (Jarque-Bera p ≈ 0) is observed across frequencies but is expected given the presence of price spikes and supply disruptions in the sample period — including COVID-19 pandemic effects and post-Brexit trade adjustments — and does not invalidate the Granger test results given the large sample sizes available at weekly and monthly frequencies.

Limitations

Several limitations should be noted. First, Granger causality is a test of predictive precedence, not structural causation — significant results indicate timing relationships but not underlying mechanisms. Second, the quarterly analysis suffers from low statistical power (n = 45) and stationarity test results at this frequency should be interpreted cautiously, as the ADF test has well-known size distortions in small samples. Third, the fertiliser series covers only 2017 onward, limiting the power of tests involving this variable. Fourth, the analysis assumes linear, time-invariant relationships — structural breaks associated with Brexit and the COVID-19 pandemic may violate this assumption, and rolling-window Granger tests could be a productive extension. Finally, the LOCF treatment of quarterly SPPI data introduces artificial constancy at weekly and monthly frequencies; a quarterly-only analysis of SPPI’s relationship with price is more methodologically defensible.

Summary

CPI is the most robust leading indicator of UK fruit and vegetable price movements, with significant pairwise Granger causality at both monthly (p = 0.038) and quarterly (p = 0.004) frequencies. Fuel price shows a weak, transient short-run effect at the weekly level only, while SPPI shows a borderline monthly signal. API and fertiliser prices exhibit no significant predictive relationship with food prices at any frequency — a notable null result given the economic prior. Joint VAR tests are non-significant across all frequencies, reflecting reduced power in the presence of multiple null predictors, and the weekly joint test is inconclusive due to matrix singularity. Taken together, the results point to demand-side inflation as the primary macroeconomic driver of fruit and vegetable price dynamics in the UK, with supply-side cost factors playing a less discernible role within this analytical framework.
